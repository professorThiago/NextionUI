635 class NexRtc {
636 public:
647  void escrever(uint16_t ano, uint8_t mes, uint8_t dia,
648  uint8_t hora, uint8_t min, uint8_t seg)
649  {
650  nexEnviarCmdF("rtc0=%u", ano);
651  nexEnviarCmdF("rtc1=%u", mes);
652  nexEnviarCmdF("rtc2=%u", dia);
653  nexEnviarCmdF("rtc3=%u", hora);
654  nexEnviarCmdF("rtc4=%u", min);
655  nexEnviarCmdF("rtc5=%u", seg);
656  }
657 
666  bool ler(const char* campo, uint32_t& dest)
667  {
668  const char* campos[] = {"ano","mes","dia","hora","min","seg","semana",nullptr};
669  const char* rtcIds[] = {"0","1","2","3","4","5","6"};
670  for (uint8_t i = 0; campos[i]; i++) {
671  if (strcmp(campo, campos[i]) == 0) {
672  nexEnviarCmdF("get rtc%s", rtcIds[i]);
673  return nexLerNumero(&dest);
674  }
675  }
676  return false;
677  }
678 
688  bool lerHora(uint32_t& hora, uint32_t& min, uint32_t& seg)
689  {
690  return ler("hora", hora) && ler("min", min) && ler("seg", seg);
691  }
692 };
